\textbf{Pro Tips:}
\begin{itemize}
\item Keep \texttt{inputf.in} and your \texttt{.cpp} file in same folder
\item The executable is created without extension (Linux)
\item For Windows, change to: \texttt{g++.exe}, \texttt{.exe} extension, and remove \texttt{./}
\item I/O redirection: \texttt{<inputf.in>outputf.out}
\end{itemize}

\textbf{Windows Adaptation:}
\begin{verbatim}
{
    "cmd": ["g++.exe", "-std=c++17", "${file}",
            "-o", "${file_base_name}.exe",
            "&&", "${file_base_name}.exe<inputf.in>outputf.out"],
    "shell": true,
    "working_dir": "$file_path",
    "selector": "source.cpp"
}
\end{verbatim}
